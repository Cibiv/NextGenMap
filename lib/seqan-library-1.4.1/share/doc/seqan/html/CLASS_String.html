Class
String
A sequence container with generic alphabet and many specializations.
String is at the heart of the SeqAn library (SeqAn is for SEQuence ANalaysis after all). There are various specializations with Alloc String being the default and most widely used one. Strings can be used to store arbitrary values and can be used for large biologicaly sequences as well as a generic, dynamic array and replace std::vector<>.

Examples
This example shows a brute force pattern matching scheme for two character Strings. Creation of String "text" shows the usage of some available String operating functions. See class StringSet for an example of a String container with other than simple type values. See class Index example for efficiently finding the same pattern matches using an index.

4using namespace seqan;
5int main()

7    // Creating text

9    std::cout << text << std::endl;
10    appendValue(text, ' ');
11    std::cout << "Last sign is whitespace? " << endsWith(text, ' ') << std::endl;
12    // Erasing whitespaces in text
13    eraseBack(text);
14    erase(text, 2);
15    // Appending another string
16    append(text, "ornottobe");
17    std::cout << text << std::endl;
18
19    // Pattern
20    String<char> pattern = "be";
21    // Number of consecutive matching characters per position
22    String<int> score;
23    resize(score, length(text) - length(pattern) + 1);
24
25    // Brute force pattern matching for every position
26    for (unsigned i = 0; i < length(text) - length(pattern) + 1; ++i)
27    {
28        int localScore = 0;
29        for (unsigned j = 0; j < length(pattern); ++j)
30            if (text[i + j] == pattern[j])
31                ++localScore;
32        score[i] = localScore;
33    }
34
35    std::cout << "hit at ";
36    for (unsigned i = 0; i < length(score); ++i)
37        if (score[i] == length(pattern))
38            std::cout << i << " ";
39    std::cout << std::endl;    
40
41    return 0;
42}
